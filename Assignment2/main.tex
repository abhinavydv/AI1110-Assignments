    \textbf{ICSE class 12 paper 2019 Q15 (c)}\\
    Find the angle between the two lines $2x = 3y = -z$ and $6x = -y = -4z$.\\\\
    \solution\\
    The two lines can be expressed in vector form as
    \begin{align}
        \label{eq:line1}
        \vec{r} & = \lambda \myvec{3 \\ 2 \\ -6}\\
        \label{eq:line2}
        \vec{r} & = \mu \myvec{2 \\ -12 \\ -3}
    \end{align}

    Let
    \begin{align}
        \vec{m} & = \myvec{3 \\ 2 \\ -6}\\
        \vec{n} & = \myvec{2 \\ -12 \\ -3}
    \end{align}

    \begin{figure}[ht!]
        \includegraphics[width=\columnwidth]{figs/plot.png}
        \caption{3D plot showing the two lines}
        \label{fig:fig1}
    \end{figure}

    The angle between the two lines is given by\\
    \begin{align}
        \theta & = \arccos\brak{\frac{\vec{m}^{\top} \vec{n}}{\norm{\vec{m}} \norm{\vec{n}}}}\\
        \implies \theta & = \arccos\brak{\frac{\myvec{3 & 2 & -6}^{\top} \myvec{2 \\ -12 \\ -3}}
                            {\norm{\myvec{3 \\ 2 \\ -6}} \norm{\myvec{2 \\ -12 \\ -3}}}}\\
        \implies \theta & = \arccos(0)\\
        \implies \theta & = \frac{\pi}{2}
    \end{align}
